1//! Managing delta exceptions.
2//!
3//! Implements 6 instructions.
4//!
5//! See <https://learn.microsoft.com/en-us/typography/opentype/spec/tt_instructions#managing-exceptions>
6
7use super::{super::graphics::CoordAxis, Engine, F26Dot6, OpResult};
8use read_fonts::tables::glyf::bytecode::Opcode;
9
10impl Engine<'_> {
11    /// Delta exception P1, P2 and P3.
12    ///
13    /// DELTAP1[] (0x5D)
14    /// DELTAP2[] (0x71)
15    /// DELTAP3[] (0x72)
16    ///
17    /// Pops: n: number of pairs of exception specifications and points (uint32)
18    ///       p1, arg1, p2, arg2, ..., pnn argn: n pairs of exception specifications
19    ///             and points (pairs of uint32s)
20    ///
21    /// DELTAP moves the specified points at the size and by the
22    /// amount specified in the paired argument. An arbitrary number of points
23    /// and arguments can be specified.
24    ///
25    /// The only difference between the instructions is the bias added to the
26    /// point adjustment.
27    ///
28    /// See <https://learn.microsoft.com/en-us/typography/opentype/spec/tt_instructions#delta-exception-p1>
29    /// and <https://gitlab.freedesktop.org/freetype/freetype/-/blob/57617782464411201ce7bbc93b086c1b4d7d84a5/src/truetype/ttinterp.c#L6509>
30    pub(super) fn op_deltap(&mut self, opcode: Opcode) -> OpResult {
31        let gs = &mut self.graphics;
32        let ppem = gs.ppem as u32;
33        let n = self.value_stack.pop_usize()?;
34        let bias = match opcode {
35            Opcode::DELTAP2 => 16,
36            Opcode::DELTAP3 => 32,
37            _ => 0,
38        } + gs.delta_base as u32;
39        let back_compat = gs.backward_compatibility;
40        let did_iup = gs.did_iup_x && gs.did_iup_y;
41        for _ in 0..n {
42            let point_ix = self.value_stack.pop_usize()?;
43            let mut b = self.value_stack.pop()?;
44            // FreeType notes that some popular fonts contain invalid DELTAP
45            // instructions so out of bounds points are ignored.
46            // See <https://gitlab.freedesktop.org/freetype/freetype/-/blob/57617782464411201ce7bbc93b086c1b4d7d84a5/src/truetype/ttinterp.c#L6537>
47            if point_ix >= gs.zp0().points.len() {
48                continue;
49            }
50            let mut c = (b as u32 & 0xF0) >> 4;
51            c += bias;
52            if ppem == c {
53                // Blindly copying FreeType here
54                // <https://gitlab.freedesktop.org/freetype/freetype/-/blob/57617782464411201ce7bbc93b086c1b4d7d84a5/src/truetype/ttinterp.c#L6565>
55                b = (b & 0xF) - 8;
56                if b >= 0 {
57                    b += 1;
58                }
59                b *= 1 << (6 - gs.delta_shift as i32);
60                let distance = F26Dot6::from_bits(b);
61                if back_compat {
62                    if !did_iup
63                        && ((gs.is_composite && gs.freedom_vector.y != 0)
64                            || gs.zp0().is_touched(point_ix, CoordAxis::Y)?)
65                    {
66                        gs.move_point(gs.zp0, point_ix, distance)?;
67                    }
68                } else {
69                    gs.move_point(gs.zp0, point_ix, distance)?;
70                }
71            }
72        }
73        Ok(())
74    }

76    /// Delta exception C1, C2 and C3.
77    ///
78    /// DELTAC1[] (0x73)
79    /// DELTAC2[] (0x74)
80    /// DELTAC3[] (0x75)
81    ///
82    /// Pops: n: number of pairs of exception specifications and CVT entry numbers (uint32)
83    ///       c1, arg1, c2, arg2,..., cn, argn: (pairs of uint32s)
84    ///
85    /// DELTAC changes the value in each CVT entry specified at the size and
86    /// by the amount specified in its paired argument.
87    ///
88    /// The only difference between the instructions is the bias added to the
89    /// adjustment.
90    ///
91    /// See <https://learn.microsoft.com/en-us/typography/opentype/spec/tt_instructions#delta-exception-c1>
92    /// and <https://gitlab.freedesktop.org/freetype/freetype/-/blob/57617782464411201ce7bbc93b086c1b4d7d84a5/src/truetype/ttinterp.c#L6604>
93    pub(super) fn op_deltac(&mut self, opcode: Opcode) -> OpResult {
94        let gs = &mut self.graphics;
95        let ppem = gs.ppem as u32;
96        let n = self.value_stack.pop_usize()?;
97        let bias = match opcode {
98            Opcode::DELTAC2 => 16,
99            Opcode::DELTAC3 => 32,
100            _ => 0,
101        } + gs.delta_base as u32;
102        for _ in 0..n {
103            let cvt_ix = self.value_stack.pop_usize()?;
104            let mut b = self.value_stack.pop()?;
105            let mut c = (b as u32 & 0xF0) >> 4;
106            c += bias;
107            if ppem == c {
108                // See <https://gitlab.freedesktop.org/freetype/freetype/-/blob/57617782464411201ce7bbc93b086c1b4d7d84a5/src/truetype/ttinterp.c#L6660>
109                b = (b & 0xF) - 8;
110                if b >= 0 {
111                    b += 1;
112                }
113                b *= 1 << (6 - gs.delta_shift as i32);
114                let cvt_val = self.cvt.get(cvt_ix)?;
115                self.cvt.set(cvt_ix, cvt_val + F26Dot6::from_bits(b))?;
116            }
117        }
118        Ok(())
119    }

122#[cfg(test)]
123mod tests {
124    use super::super::{super::zone::ZonePointer, MockEngine};
125    use raw::{
126        tables::glyf::bytecode::Opcode,
127        types::{F26Dot6, Point},
128    };
129
130    #[test]
131    fn deltap() {
132        let mut mock = MockEngine::new();
133        let mut engine = mock.engine();
134        engine.graphics.backward_compatibility = false;
135        engine.graphics.zp0 = ZonePointer::Glyph;
136        let raw_ppem = 16;
137        let raw_adjustment = 7;
138        for (point_ix, (ppem_bias, opcode)) in [
139            (0, Opcode::DELTAP1),
140            (16, Opcode::DELTAP2),
141            (32, Opcode::DELTAP3),
142        ]
143        .iter()
144        .enumerate()
145        {
146            let ppem = raw_ppem + ppem_bias;
147            engine.graphics.ppem = ppem;
148            // packed ppem + adjustment entry
149            let packed_ppem = raw_ppem - engine.graphics.delta_base as i32;
150            engine
151                .value_stack
152                .push((packed_ppem << 4) | raw_adjustment)
153                .unwrap();
154            // point index
155            engine.value_stack.push(point_ix as _).unwrap();
156            // exception count
157            engine.value_stack.push(1).unwrap();
158            engine.op_deltap(*opcode).unwrap();
159            let point = engine.graphics.zones[1].point(point_ix).unwrap();
160            assert_eq!(point.map(F26Dot6::to_bits), Point::new(-8, 0));
161        }
162    }
